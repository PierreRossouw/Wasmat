func i32 main() export
  bool isPrime = true
  i32 numPrimes = 0
  i32 i = 100000
  loop
    breakif i <= 1
    i32 j = 2
    loop
      breakif j >= i
      if i mod j == 0 then
        isPrime = false
        break
      end
      j = j + 1
    end
    if isPrime then
      numPrimes = numPrimes + 1
    end
    isPrime = true
    i = i - 1
  end
  return numPrimes
end
